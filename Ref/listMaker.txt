export function makeList(data) {

    // Mapの場合
    if (data instanceof Map) {
        data.forEach((value, key) => {
            console.log('k=' + key);
            console.log('v=' + value);
        });
        return;
    }

    // 配列 + オブジェクトの場合
    data.forEach((el) => {
        Object.entries(el).forEach(([k, v]) => {
            console.log('k=' + k);
            console.log('v=' + v);
        });
    });
}

li.textContent = k+v　　で変数vのみ非表示にする方法は

結論から言うと👇

👉 textContent を使う限り「部分的に非表示」はできません

🔴 なぜ？
li.textContent = k + v;

👉 これは「ただの文字列」になる
👉 "ax" みたいに1つのテキストとして扱われる

→ vだけ隠すことは不可能

✅ 解決方法（要素を分ける）
✔ 方法①：spanで分ける（おすすめ）
const li = document.createElement('li');

const keySpan = document.createElement('span');
keySpan.textContent = k;

const valueSpan = document.createElement('span');
valueSpan.textContent = v;
valueSpan.style.display = 'none'; // ← 非表示

li.appendChild(keySpan);
li.appendChild(valueSpan);

👉 これで vだけ非表示 にできる

✔ 表示/非表示を切り替える
valueSpan.style.display = 'inline'; // 表示
valueSpan.style.display = 'none';   // 非表示
✔ 方法②：classで制御（実用）
valueSpan.classList.add('hidden');
